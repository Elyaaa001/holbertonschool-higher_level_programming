# Handling invalid instantiation cases
my_list = MyList(None)  # Raises TypeError: 'NoneType' object is not iterable

my_list = MyList([1, 2], [3, 4])  # Raises TypeError: list expected at most 1 argument, got 2

# Valid instantiation and sorted printing
my_list = MyList([1, 2, 3])
my_list.print_sorted()  # Output: [1, 2, 3]

# Mixed types case
my_list = MyList([1, "Betty", "Holberton", 5])
my_list.print_sorted()  # Raises TypeError: unorderable types: cannot compare different types in the list
